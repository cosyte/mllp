HL7 v2 MSH/MSA segment positional reference:
- MSH-10 (Message Control ID): segment is `MSH|^~\&|...|...|...|...|YYYYMMDDHHmmSS||TYPE^TRIGGER^STRUCTURE|<MSH-10>|...`
  - Field separator is the byte immediately AFTER the literal `MSH` (typically `|` = 0x7C, but ANY byte is legal — read MSH[3] of the payload to determine).
  - Field 10 of the FIRST segment (MSH).
- MSA-2 (Message Control ID being acknowledged): `MSA|<MSA-1>|<MSA-2>|...`
  - In an ACK message, MSA-2 echoes the inbound MSH-10.
  - Find the MSA segment by scanning segments for one starting with `MSA` followed by the field separator.

Both extractors are PURE BYTE/STRING operations — never construct a parser, never depend on `@cosyte/hl7`.

       const ASCII_M = 0x4d, ASCII_S = 0x53, ASCII_H = 0x48, ASCII_A = 0x41;
       const SEGMENT_SEPARATOR_CR = 0x0d;
       const SEGMENT_SEPARATOR_LF = 0x0a;

       /**
        * Extract MSH-10 (Message Control ID) from an HL7 v2 payload.
        *
        * Pure byte-level scan — never throws, returns `null` for malformed input
        * (Postel's Law decoder side; CLAUDE.md guardrail).
        *
        * @internal
        */
       export function extractMshControlId(buf: Buffer): string | null {
         if (buf.length < 4) return null;
         if (buf[0] !== ASCII_M || buf[1] !== ASCII_S || buf[2] !== ASCII_H) return null;
         const fieldSep = buf[3] as number;
         // MSH treats positions 3 (sep) and 4 (encoding chars) specially.
         // MSH segment fields when split by `fieldSep`: ['MSH', encChars, MSH-3, MSH-4, ..., MSH-10, ...]
         // Index 0 = 'MSH', Index 1 = encoding chars, Index 2..N = MSH-3..MSH-(N+1).
         // So MSH-10 is at split-array index 9.
         //
         // Iterate byte-by-byte: count separators and capture range for the 9th split.
         let fieldIndex = 0;            // 0 maps to 'MSH'
         let fieldStart = 0;
         const end = buf.length;
         for (let i = 3; i < end; i++) {   // start at the byte AFTER 'MSH', which is the first separator
           const b = buf[i];
           if (b === fieldSep || b === SEGMENT_SEPARATOR_CR || b === SEGMENT_SEPARATOR_LF) {
             fieldIndex++;
             if (fieldIndex === 9) fieldStart = i + 1;
             else if (fieldIndex === 10) {
               if (fieldStart >= i) return null;   // empty MSH-10
               return buf.subarray(fieldStart, i).toString('ascii');
             }
             if (b === SEGMENT_SEPARATOR_CR || b === SEGMENT_SEPARATOR_LF) return null;  // segment ended before MSH-10
           }
         }
         return null;
       }

       /**
        * Extract MSA-2 (acknowledged Message Control ID) from an HL7 v2 ACK payload.
        *
        * @internal
        */
       export function extractMsaControlId(buf: Buffer): string | null {
         // Find a segment starting with bytes 'MSA' followed by the same field separator used in MSH.
         if (buf.length < 4) return null;
         const fieldSep = buf[3] as number;     // MSH establishes the separator
         const end = buf.length;
         // Scan for segment-start of "MSA<sep>"
         let segStart = 0;
         while (segStart < end) {
           // Find segment terminator
           let segEnd = segStart;
           while (segEnd < end && buf[segEnd] !== SEGMENT_SEPARATOR_CR && buf[segEnd] !== SEGMENT_SEPARATOR_LF) segEnd++;
           if (segEnd - segStart >= 4 &&
               buf[segStart] === ASCII_M && buf[segStart + 1] === ASCII_S && buf[segStart + 2] === ASCII_A &&
               buf[segStart + 3] === fieldSep) {
             // MSA segment: split between segStart+3 (first sep) and segEnd
             // Fields: [0]'MSA', [1]MSA-1, [2]MSA-2, ...
             let fieldIndex = 0;
             let fieldStart = 0;
             for (let i = segStart + 3; i <= segEnd; i++) {
               const b = i < segEnd ? (buf[i] as number) : fieldSep;   // sentinel for last field
               if (b === fieldSep || b === SEGMENT_SEPARATOR_CR || b === SEGMENT_SEPARATOR_LF) {
                 fieldIndex++;
                 if (fieldIndex === 2) fieldStart = i + 1;
                 else if (fieldIndex === 3) {
                   if (fieldStart >= i) return null;    // empty MSA-2
                   return buf.subarray(fieldStart, i).toString('ascii');
                 }
               }
             }
             return null;
           }
           // Skip past terminator(s) to next segment start
           while (segEnd < end && (buf[segEnd] === SEGMENT_SEPARATOR_CR || buf[segEnd] === SEGMENT_SEPARATOR_LF)) segEnd++;
           if (segEnd === segStart) return null;
           segStart = segEnd;
         }
         return null;
       }
