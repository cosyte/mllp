    Round-trip fidelity (FRAME-12):
    - encodeFrame + FrameReader round-trip for every byte value 0x00-0xFF:
      - For bytes NOT equal to VT (0x0B) or FS (0x1C): use strict encodeFrame (no opts), all bytes preserve
      - For VT (0x0B) and FS (0x1C): use encodeFrame with allowDelimiterBytesInPayload: true, then decode with no tolerance opts (the decoder just reads whatever is between VT and FS+CR — the issue is the ENCODER that might corrupt framing; the decoder receives the raw bytes verbatim since the encoder passes them through with the flag)
        Wait — this is subtle: if allowDelimiterBytesInPayload: true encodes a payload containing VT bytes verbatim, then the encoded frame contains a VT byte INSIDE the payload. The decoder in SCANNING_FOR_VT state will treat that VT as a new frame start, not as a payload byte.
        The round-trip test for VT/FS bytes needs to use the decoder with matching tolerances to correctly handle this case, OR use a different payload construction: encode the single-byte payload [0x0B], then push the resulting frame through the decoder. Since the frame is [VT=0x0B, 0x0B, FS=0x1C, CR=0x0D], the decoder in READING_PAYLOAD sees byte 0x0B = VT and treats it as VT mid-payload → MLLP_TRAILING_BYTES + reset.
        The actual semantics: VT and FS bytes IN a payload are genuinely ambiguous at the wire level. FRAME-02/FRAME-12 says these bytes "must be handled by FRAME-02" — meaning the VT/FS test covers that the ENCODER correctly detects and throws/warns, not that they survive a naive round-trip.
        For the byte fidelity test: test all bytes EXCEPT 0x0B and 0x1C through strict round-trip (254 values). For 0x0B and 0x1C, test that encodeFrame throws MllpFramingError (strict path) — that IS the correct behavior per FRAME-02.
    - 254-byte test: each byte 0x00-0xFF except 0x0B and 0x1C — encodeFrame then decode → same byte
    - 0x0B and 0x1C: encodeFrame without opts → throws MllpFramingError (validates FRAME-02 guard works)
    - 1 MB random corpus: generate 1 MiB of random bytes, remove all 0x0B and 0x1C bytes, encode, decode via FrameReader, decoded payload === original bytes
    - Chunked delivery: push corpus frame in 1-byte chunks → same result
  </behavior>

    import { describe, it, expect } from 'vitest';
    import { encodeFrame } from '../../src/framing/encoder.js';
    import { FrameReader } from '../../src/framing/decoder.js';
    import { MllpFramingError } from '../../src/framing/error.js';

    const VT = 0x0b;
    const FS = 0x1c;

    /**
     * Round-trip helper: encode a payload and decode it via FrameReader.
     * Returns the decoded payload buffer.
     */
    function roundTrip(payload: Buffer): Buffer {
      let received: Buffer | undefined;
      const r = new FrameReader({
        onFrame: (p) => { received = p; },
      });
      const encoded = encodeFrame(payload);
      r.push(encoded);
      if (received === undefined) throw new Error('No frame delivered');
      return received;
    }

    /**
     * Round-trip in 1-byte chunks.
     */
    function roundTripChunked(payload: Buffer): Buffer {
      let received: Buffer | undefined;
      const r = new FrameReader({ onFrame: (p) => { received = p; } });
      const encoded = encodeFrame(payload);
      for (let i = 0; i < encoded.length; i++) {
        r.push(encoded.subarray(i, i + 1));
      }
      if (received === undefined) throw new Error('No frame delivered (chunked)');
      return received;
    }

    describe('FRAME-12: byte-fidelity round-trip', () => {
      it('every byte value 0x00-0xFF except 0x0B/0x1C round-trips unchanged', () => {
        for (let b = 0; b <= 0xff; b++) {
          if (b === VT || b === FS) continue; // handled in next test
          const payload = Buffer.from([b]);
          const decoded = roundTrip(payload);
          expect(decoded[0]).toBe(b);
          expect(decoded.length).toBe(1);
        }
      });

      it('0x0B (VT) in payload causes MllpFramingError (FRAME-02 guard)', () => {
        expect(() => encodeFrame(Buffer.from([VT]))).toThrow(MllpFramingError);
        try {
          encodeFrame(Buffer.from([VT]));
        } catch (err) {
          expect((err as MllpFramingError).code).toBe('MLLP_PAYLOAD_CONTAINS_VT');
        }
      });

      it('0x1C (FS) in payload causes MllpFramingError (FRAME-02 guard)', () => {
        expect(() => encodeFrame(Buffer.from([FS]))).toThrow(MllpFramingError);
        try {
          encodeFrame(Buffer.from([FS]));
        } catch (err) {
          expect((err as MllpFramingError).code).toBe('MLLP_PAYLOAD_CONTAINS_FS');
        }
      });

      it('multi-byte payload with all safe bytes round-trips unchanged', () => {
        // Build a payload with all safe bytes (excluding VT/FS)
        const safeBytes: number[] = [];
        for (let b = 0; b <= 0xff; b++) {
          if (b !== VT && b !== FS) safeBytes.push(b);
        }
        const payload = Buffer.from(safeBytes);
        const decoded = roundTrip(payload);
        expect(decoded).toEqual(payload);
      });

      it('1 MiB random corpus (excluding VT/FS bytes) round-trips unchanged', () => {
        // Generate 1 MiB of deterministic pseudo-random bytes, filtering out VT and FS
        const SIZE = 1024 * 1024;
        const raw = Buffer.allocUnsafe(SIZE);
        // Use a simple LCG for determinism (no crypto dependency)
        let seed = 0xdeadbeef;
        for (let i = 0; i < SIZE; i++) {
          seed = (seed * 1664525 + 1013904223) >>> 0;
          raw[i] = seed & 0xff;
        }
        // Strip VT and FS bytes (replace with 0x41 'A')
        for (let i = 0; i < raw.length; i++) {
          const b = raw[i];
          if (b === VT || b === FS) raw[i] = 0x41;
        }
        const decoded = roundTrip(raw);
        expect(decoded).toEqual(raw);
        expect(decoded.length).toBe(SIZE);
      });

      it('1 MiB corpus round-trips correctly in 1-byte chunks', () => {

          payload[i] = (b === VT || b === FS) ? 0x41 : b;
        }
        const decoded = roundTripChunked(payload);
        expect(decoded).toEqual(payload);
      });
    });
